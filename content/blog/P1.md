# 2.环境渐变色
 绿通道是G，代表上方向，取负代表下方向，用1（白色）减去向下的方向就得到其他方向的叠加（侧两面）。得到各方向的遮罩。
对上，下，侧方向的遮罩乘以颜色然后再混合，即add。 得到环境渐变。最后再multply环境AO阴影。
代码实现更简洁。
```HLSL
//混合运算
                float3 envCol = _UpCol * upMask + _DownCol * downMask + _MidCol * sideMask;
                float occlusion = tex2D(_AOtex,i.uv);               //采样贴图函数tex2D
                occlusion = lerp(1.0, occlusion, _AOIntensity); 
                float3 envLit = envCol * occlusion;
                //返回结果
                return float4(envLit,1.0);         
```
完整代码：
```HLSL
Shader "L0/L07/EnvColor_CODE" {                 //路径名 
    Properties {                        //unity里inspector的参数
    ///格式。_名字 (“面板名字”，类型)=默认值，并且VertexInput之前也要声明，或之前声明即可
    //起手下划线加大写
    _AOIntensity ("AO强度", Range(0, 20)) = 0.8 
    _AOtex ("AO图",2d) = "white"{}
    _UpCol ("上环境",color) = (1.0,1.0,1.0,1.0)
    _MidCol ("中环境",color) = (0.5,0.5,0.5,1.0)
    _DownCol ("下环境",color) = (0.0,0.0,0.0,1.0)

    }


    SubShader {
        Tags {
            "RenderType"="Opaque"
        }
        Pass {
            Name "FORWARD"
            Tags {
                "LightMode"="ForwardBase"
            }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            #include "UnityPBSLighting.cginc"
            #include "UnityStandardBRDF.cginc"
            #pragma multi_compile_fwdbase_fullshadows
            #pragma target 3.0

            //声明参数
            uniform float3 _UpCol;
            uniform float3 _MidCol;
            uniform float3 _DownCol;
            uniform float _AOIntensity;
            uniform sampler2D _AOtex;
                                                                    //模型的相关信息输入
            struct VertexInput {                                    //输入结构
                float4 vertex : POSITION;                           //顶点信息
                float3 normal : NORMAL;                             //法线信息
                float2 uv0 : TEXCOORD0;                             //uv信息：有贴图就要有uv，0通道，一共有4个通道
            };


            struct VertexOutput {                                   //输出结构
                float4 pos : SV_POSITION;                           //裁切空间，可以先理解为屏幕空间
                float3 nDirWS : TEXCOORD0;                          //世界 物体法线位置
                float2 uv : TEXCOORD1;                              //世界 顶点位置
            };

            //输入》顶点Shader》输出
            VertexOutput vert (VertexInput v) {                     //顶点Shader
                VertexOutput o = (VertexOutput)0;
                o.pos = UnityObjectToClipPos( v.vertex );         //映射到屏幕空间的位置，Unity提供的方法
                o.nDirWS = UnityObjectToWorldNormal(v.normal);      //本地转世界空间，Unity提供的方法
                o.uv = v.uv0;                                       //uv采样，输出uv等于输入uv0
                return o;
            }

            //输出》像素shader
            float4 frag(VertexOutput i) : COLOR {                   //像素shader
                //准备向量
                float3 nDir = i.nDirWS;                             //在像素shader叫i，在顶点叫o
                
                //准备结果
                float upMask = max(0.0 , nDir.g);
                float downMask = max(0.0 , -nDir.g);
                float sideMask = 1.0 - upMask - downMask;

                //混合运算
                float3 envCol = _UpCol * upMask + _DownCol * downMask + _MidCol * sideMask;
                float occlusion = tex2D(_AOtex,i.uv);               //采样贴图函数tex2D
                occlusion = lerp(1.0, occlusion, _AOIntensity); 
                float3 envLit = envCol * occlusion;
                //返回结果
                return float4(envLit,1.0);         //规范化，输出lambert1
            }






            ENDCG
        }
    }
    FallBack "Diffuse"

    CustomEditor "ShaderForgeMaterialInspector"
}
```

```
                float4 pos : SV_POSITION;                         //裁切空间，可以先理解为屏幕空间
                LIGHTING_COORDS(0,1)                                //投影坐标，Unity已经封装的，格式有点奇怪，如果占用了COORD0那么就填（1，2）

                return o;
            }

            //输出》像素shader
            float4 frag(VertexOutput i) : COLOR {                   //像素shader
                float shadow = LIGHT_ATTENUATION(i);                //unity封装，可取出投影
                //返回结果
                return float4(shadow,shadow,shadow,1.0);
            }






            ENDCG
        }
    }
    FallBack "Diffuse"

    CustomEditor "ShaderForgeMaterialInspector"
```

# 4.综合代码（直接光间接光阴影色）
![S6](./assets/Blog/P1/S6.png)
```HLSL
Shader "L0/L07/OSC_Plus" {                 //路径名

    Properties {                        //unity里inspector的参数
        _MainCol ("主色", Color) = (0.0,0.0,0.0,1)
        _Occlusion ("AO遮罩", 2D) = "white" {}
        _AOIntensity ("AO强度", Range(0, 1)) = 0
        _UpColor ("上环境色", Color) = (0.7,0.9,1,1)
        _SideColor ("侧环境色", Color) = (0.5,0.7,0.5,1)
        _DownColor ("下环境色", Color) = (0.6,0.5,0.4,1)
        _HighLight ("亮纹理强度", Range(0, 100)) = 0
        _HighLightTEX ("高光纹理", 2D) = "white" {}
        _HighLightBLACK ("暗纹理强度", Range(0, 10)) = 1
        _ShaderColor ("阴影颜色", Color) = (0.5,0.5,0.5,0.7)
    }


    SubShader {
        Tags {
            "RenderType"="Opaque"
        }
        Pass {
            Name "FORWARD"
            Tags {
                "LightMode"="ForwardBase"
            }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            #include "AutoLight.cginc"  //unity投影必须有这个库
            #include "Lighting.cginc"   //同上

            #pragma multi_compile_fwdbase_fullshadows
            #pragma target 3.0




            //声明参数
            uniform float3 _MainCol;
            uniform float3 _UpColor;
            uniform sampler2D _Occlusion;
            uniform float3 _SideColor;
            uniform float3 _DownColor;
            uniform float _HighLight;
            uniform float _AOIntensity;
            uniform sampler2D _HighLightTEX;
            uniform float _HighLightBLACK;
            uniform float3 _ShaderColor;

            struct VertexInput {                                    //输入结构
                float4 vertex : POSITION;
                float4 normal : NORMAL;
                float2 uv0    : TEXCOORD0;

            };


            struct VertexOutput {                                   //输出结构
                float4 pos : SV_POSITION;                           //裁切空间，可以先理解为屏幕空间
                float2 uv0 : TEXCOORD0;
                float3 posWS : TEXCOORD1;
                float3 nDirWS : TEXCOORD2;
                LIGHTING_COORDS(3,4)                                //投影坐标，Unity已经封装的，格式有点奇怪，如果占用了COORD0那么就填（1，2）

            };

            //输入》顶点Shader》输出
            VertexOutput vert (VertexInput v) {                     //顶点Shader
                VertexOutput o = (VertexOutput)0;
                o.pos = UnityObjectToClipPos( v.vertex );           //映射到屏幕空间的位置，Unity提供的方法
                o.uv0 = v.uv0;
                o.posWS = mul(unity_ObjectToWorld,v.vertex);
                o.nDirWS = UnityObjectToWorldNormal(v.normal);
                TRANSFER_VERTEX_TO_FRAGMENT(o);                     //unity封装，实现投影，顶点shader必须加。TRANSFER_VERTEX_TO_FRAGMENT 宏强制要求结构体有 pos 字段（名称必须是 pos，不能改）
                
                return o;
            }

            //输出》像素shader
            float4 frag(VertexOutput i) : COLOR {                   //像素shader
                //准备向量
                float3 nDir = i.nDirWS;                             //在像素shader叫i，在顶点叫o
                float3 lDir = _WorldSpaceLightPos0.xyz;

                float3 vDir = normalize(_WorldSpaceCameraPos.xyz - i.posWS); //相机位置 - 顶点位置，等于视角方向，初始位置(顶点)指向终点位置，再归一化
                float3 hDir = normalize(vDir + lDir);               //半角，向量的合成，相加
                
                //准备结果
                float ndotl = dot(nDir,lDir);                       //初始操作，shadergraph第一步连连看，兰伯特
                float ndoth = dot(nDir,hDir);                       //BlinnPhong，类似兰伯特，一个截断加一个power突出高光
                float upMask = max(0.0 , nDir.g);
                float downMask = max(0.0 , -nDir.g);
                float sideMask = 1.0 - upMask - downMask;

                //光照模型(直接光照)
                float lambert = max(0.0,ndotl);
                lambert = lambert * 0.5 + 0.5;
                //float blinnPhong = pow(max(0.0,ndoth),_SpeculPow);
                //float occlusion = tex2D(_AOtex,i.uv);
                float shadow = LIGHT_ATTENUATION(i);                //unity封装，可取出投影
                float highlitTEX = tex2D(_HighLightTEX,i.uv0).r;
                float specPow = lerp(_HighLightBLACK, _HighLight, highlitTEX);
                float blinnPhong = pow(max(0.0,ndoth),specPow);
                float3 dirLighting = _MainCol * lambert + blinnPhong;  //规范，先把3维放前面乘，加上blinnPhong可以只把高光混合进来因为黑值是0.


                //环境模型(间接光照)
                float3 envCol = _UpColor * upMask + _DownColor * downMask + _SideColor * sideMask;
                float occlusion = tex2D(_Occlusion,i.uv0);               //采样贴图函数tex2D
                occlusion = lerp(1.0, occlusion, _AOIntensity); 
                float3 envLit = envCol * occlusion;

                //阴影颜色混合
                dirLighting = dirLighting * envLit;
                float3 finalRGB = lerp(_ShaderColor,dirLighting,shadow);

                //返回结果
                return float4(finalRGB,1.0);


            }






            ENDCG
        }
    }
    FallBack "Diffuse"

    CustomEditor "ShaderForgeMaterialInspector"
}
```

# 5.法线
![S7](./assets/Blog/P1/S7.png)
1.纹理法线需要解包才能用
2.TBN 矩阵是 “纹理空间→世界空间” 的转换矩阵（即切线空间的法线信息转到世界空间）
3.把切线空间转到世界空间得到有凹凸法线的向量信息。
>TBN是从切线空间转世界空间 Object2World是从模型空间转世界空间

# 7.Matcap
## 一般用于模拟反射/不真实但性能好/会随镜头变化而发生变化
法线贴图是怎么变成向量信息的
>RGB映射XYZ（向量信息）
·  颜色值（如 128）先归一化为 0.0 ~ 1.0 的浮点数（128/255 ≈ 0.502，近似 0.5）；
·  把 0.0 ~ 1.0 的范围映射到 -1.0 ~ 1.0（公式：向量值 = (归一化颜色值 - 0.5) * 2）；
·  最终得到的向量分量（x/y/z）就是法线的单位向量（长度为 1）。

```HLSL
Shader "L0/L09/matCap" {                 //路径名

    Properties {                        //unity里inspector的参数
    ///格式。_名字 (“面板名字”，类型)=默认值，并且VertexInput之前也要声明，或之前声明即可
    //起手下划线加大写
        _NormalMap ("法线贴图", 2D) = "bump" {}
        _MatCap ("MatCap", 2D) = "white" {}
        _envSpecular ("环境镜面反射强度", Range(0, 5)) = 1
        _fnlSpecular ("菲涅尔次幂", Range(0, 100)) = 1
    }


    SubShader {
        Tags {
            "RenderType"="Opaque"
        }
        Pass {
            Name "FORWARD"
            Tags {
                "LightMode"="ForwardBase"
            }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            #include "UnityPBSLighting.cginc"
            #include "UnityStandardBRDF.cginc"
            #pragma multi_compile_fwdbase_fullshadows
            #pragma target 3.0

            //声明参数
            uniform sampler2D _NormalMap;
            uniform sampler2D _MatCap;
            uniform float _envSpecular;
            uniform float _fnlSpecular;

            struct VertexInput {                                    //输入结构
                float4 vertex : POSITION;
                float3 normal : NORMAL;
                float4 tangent : TANGENT;
                float2 uv0 : TEXCOORD0;
                //顶点，法线，切线（法线贴图必备），采样uv，
            };


            struct VertexOutput {                                   //输出结构
                float2 uv0 : TEXCOORD0;
                float4 pos : SV_POSITION;                         //裁切空间，可以先理解为屏幕空间
                float3 nDirWS : TEXCOORD1;                         //世界 物体法线位置
                float3 posWS : TEXCOORD2;                          //世界 顶点位置
                float3 tDirWS : TEXCOORD3;
                float3 bDirWS : TEXCOORD4;
                
            };

            //输入》顶点Shader》输出
            VertexOutput vert (VertexInput v) {                     //顶点Shader
                VertexOutput o = (VertexOutput)0;
                o.pos = UnityObjectToClipPos( v.vertex );         //映射到屏幕空间的位置，Unity提供的方法
                o.posWS = mul(unity_ObjectToWorld,v.vertex);        //unity_ObjectToWorld是全小写的！！
                o.nDirWS = UnityObjectToWorldNormal(v.normal);      //本地转世界空间，Unity提供的方法
                o.uv0 = v.uv0;
                o.tDirWS = normalize( mul( unity_ObjectToWorld, float4( v.tangent.xyz, 0.0 ) ).xyz );
                o.bDirWS = normalize(cross(o.nDirWS, o.tDirWS) * v.tangent.w);
                return o;
            }

            //输出》像素shader
            float4 frag(VertexOutput i) : COLOR {                   //像素shader
                //准备向量
                float3 nDirTS = UnpackNormal(tex2D(_NormalMap,i.uv0)).rgb;
                float3x3 TBN = float3x3(i.tDirWS,i.bDirWS,i.nDirWS);
                float3 nDirWS = normalize(mul(nDirTS,TBN));         //切线空间乘TBN矩阵转为世界空间
                float3 nDirVS = mul(UNITY_MATRIX_V,float4(nDirWS,0.0));//U_M_V矩阵是4维的，所以nDir强制转为4维，连连看只用了RG通道采样
                float3 vDirWS = normalize(_WorldSpaceCameraPos.xyz - i.posWS);
                //float3 vrDirWS = reflect( - vDirWS,nDirWS);

                //中间值准备
                //注意这里！！！是对视图空间的rg进行映射而不是世界空间！！
                float2 matcapUV = nDirVS.rg;//  * 0.5 + 0.5;            //对采样的法线的rg通道进行remap到0-1范围内，以此作为uv信息
                float ndotv = dot(nDirWS,vDirWS);                       //菲涅尔

                float fnl = pow((1- ndotv),_fnlSpecular);               



                //光照模型
                float3 matcap = tex2D(_MatCap,matcapUV);


                float3 finalRGB = (matcap + fnl) * _envSpecular;


                //返回结果
                return float4(finalRGB,1.0);         //规范化，输出lambert1
            }






            ENDCG
        }
    }
    FallBack "Diffuse"

    CustomEditor "ShaderForgeMaterialInspector"
}
```

# 8.CubeMap
与matcap不同，这是真实反射贴图，而不是matcap采样模拟反射纹理
>Matcap只用视图空间法线方向映射贴图//节省性能，方便风格化
>Cubemap用反射方向在在 6 张环境图上采样//更真实

```HLSL
Shader "L0/L09/CUBEMAP" {                 //路径名

    Properties {                        //unity里inspector的参数
    ///格式。_名字 (“面板名字”，类型)=默认值，并且VertexInput之前也要声明，或之前声明即可
    //起手下划线加大写
        _Occlusion ("AO贴图", 2D) = "bump" {}
        _NormalMap ("法线贴图", 2D) = "bump" {}
        _Cubemap ("CubeMap", Cube) = "_Skybox" {}
        _envSpecular ("环境镜面反射强度", Range(0, 5)) = 1
        _fnlSpecular ("菲涅尔次幂", Range(0, 100)) = 1
        _CubemapMIP ("CubeMap强度",Range(0,7)) = 1

    }


    SubShader {
        Tags {
            "RenderType"="Opaque"
        }
        Pass {
            Name "FORWARD"
            Tags {
                "LightMode"="ForwardBase"
            }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #include "UnityCG.cginc"
            #include "UnityPBSLighting.cginc"
            #include "UnityStandardBRDF.cginc"
            #pragma multi_compile_fwdbase_fullshadows
            #pragma target 3.0

            //声明参数
            uniform sampler2D _NormalMap;
            uniform samplerCUBE _Cubemap;
            uniform float _envSpecular;
            uniform float _fnlSpecular;
            uniform float _CubemapMIP;
            uniform sampler2D _Occlusion;
            struct VertexInput {                                    //输入结构
                float4 vertex : POSITION;
                float3 normal : NORMAL;
                float4 tangent : TANGENT;
                float2 uv0 : TEXCOORD0;
                //顶点，法线，切线（法线贴图必备），采样uv，
            };


            struct VertexOutput {                                   //输出结构
                float2 uv0 : TEXCOORD0;
                float4 pos : SV_POSITION;                         //裁切空间，可以先理解为屏幕空间
                float3 nDirWS : TEXCOORD1;                         //世界 物体法线位置
                float3 posWS : TEXCOORD2;                          //世界 顶点位置
                float3 tDirWS : TEXCOORD3;
                float3 bDirWS : TEXCOORD4;
                
            };

            //输入》顶点Shader》输出
            VertexOutput vert (VertexInput v) {                     //顶点Shader
                VertexOutput o = (VertexOutput)0;
                o.pos = UnityObjectToClipPos( v.vertex );         //映射到屏幕空间的位置，Unity提供的方法
                o.posWS = mul(unity_ObjectToWorld,v.vertex);        //unity_ObjectToWorld是全小写的！！
                o.nDirWS = UnityObjectToWorldNormal(v.normal);      //本地转世界空间，Unity提供的方法
                o.uv0 = v.uv0;
                o.tDirWS = normalize( mul( unity_ObjectToWorld, float4( v.tangent.xyz, 0.0 ) ).xyz );
                o.bDirWS = normalize(cross(o.nDirWS, o.tDirWS) * v.tangent.w);
                return o;
            }

            //输出》像素shader
            float4 frag(VertexOutput i) : COLOR {                   //像素shader
                //准备向量
                float3 nDirTS = UnpackNormal(tex2D(_NormalMap,i.uv0)).rgb;
                float3x3 TBN = float3x3(i.tDirWS,i.bDirWS,i.nDirWS);
                float3 nDirWS = normalize(mul(nDirTS,TBN));         //切线空间乘TBN矩阵转为世界空间
                float3 vDirWS = normalize(_WorldSpaceCameraPos.xyz - i.posWS);
                float3 vrDirWS = reflect( - vDirWS,nDirWS);

                //中间值准备
                float occlusion = tex2D(_Occlusion,i.uv0).r;
                float ndotv = dot(nDirWS,vDirWS);                       //菲涅尔

                float fnl = pow((1- ndotv),_fnlSpecular);               



                //光照模型
                float3 cubemap0 = texCUBElod(_Cubemap,float4(vrDirWS,_CubemapMIP));


                float3 finalRGB = cubemap0 * occlusion * _envSpecular + fnl;
                //返回结果
                return float4(finalRGB,1.0);         //规范化，输出lambert1
            }






            ENDCG
        }
    }
    FallBack "Diffuse"

    CustomEditor "ShaderForgeMaterialInspector"
}
```
